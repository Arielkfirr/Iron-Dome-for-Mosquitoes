  // ירי – לדוגמה כשהתקבלה פקודה מסריאל
  if (Serial.available()) {
    char command = Serial.read();
    if (command == 'f') {

      delay(500);
      shooter.write(0);   // חזרה למצב מנוחה
    }
  }

#include <Servo.h>

Servo myServo;

void setup() {
  myServo.attach(9);  // מחובר לפין 9
  Serial.begin(9600);
}

void loop() {
  if (Serial.available()) {
    char c = Serial.read();
    if (c == 'f') {
      myServo.write(90); delay(500); myServo.write(0);
    }
  }
}
